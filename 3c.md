import { useState } from "react";

export default function Form() {
  const [form, setForm] = useState({ name: "", email: "", phone: "" });

  const handle = e => setForm({ ...form, [e.target.name]: e.target.value });
  const submit = e => {
    e.preventDefault();
    alert(`${form.name} - ${form.email} - ${form.phone}`);
  };

  return (
    <form onSubmit={submit}>
      <input name="name" value={form.name} onChange={handle} placeholder="Name" />
      <input name="email" value={form.email} onChange={handle} placeholder="Email" />
      <input name="phone" value={form.phone} onChange={handle} placeholder="Phone No" />
      <button>Submit</button>
    </form>
  );
}
